/*
 * XIAO ESP32C6 — Contrôleur LED NeoPixel
 * BLE (Android) + WiFi WebSocket (iOS)
 *
 * Bibliothèques :
 *   - Adafruit NeoPixel
 *   - ESP32 Arduino core
 *   - WebSocketsServer (Links2004/arduinoWebSockets)
 *
 * Câblage :
 *   NeoPixel  → D1 (GPIO1)
 *   Bouton MODE   → D2 + GND
 *   Bouton BRIGHT → D3 + GND
 *   Batterie ADC  → A0
 */

#include <Adafruit_NeoPixel.h>
#include <BLEDevice.h>
#include <BLEServer.h>
#include <BLEUtils.h>
#include <BLE2902.h>
#include <WiFi.h>
#include <WebSocketsServer.h>

// ─── WiFi ─────────────────────────────────────────────────
const char* ssid     = "VOTRE_SSID";
const char* password = "VOTRE_MOT_DE_PASSE";

WebSocketsServer wsServer(81);
uint8_t wsClientNum = 255;

// ─── BLE NUS ──────────────────────────────────────────────
#define SERVICE_UUID   "6E400001-B5BA-F393-E0A9-E50E24DCCA9E"
#define CHAR_UUID_RX   "6E400002-B5BA-F393-E0A9-E50E24DCCA9E"
#define CHAR_UUID_TX   "6E400003-B5BA-F393-E0A9-E50E24DCCA9E"

BLEServer*         pServer  = nullptr;
BLECharacteristic* pTxChar  = nullptr;
bool bleConnected = false;
bool oldConnected = false;

// ─── LEDs ─────────────────────────────────────────────────
#define PIN_NEOPIXEL 1
#define NUM_LEDS     28
#define LEDS_ACTIVE  9
Adafruit_NeoPixel strip(NUM_LEDS, PIN_NEOPIXEL, NEO_GRB + NEO_KHZ800);

uint8_t currentMode = 0;
uint8_t brightness  = 180;
uint8_t pastelR[4] = {  4,  35, 100, 180 };
uint8_t pastelG[4] = { 85,  80,  60,  40 };
uint8_t pastelB[4] = { 42, 160, 120,  20 };

bool   cmdPending = false;
String pendingCmd = "";

// ─── Envoi unifié BLE + WS ────────────────────────────────
void sendResponse(String msg) {
  if (pTxChar && bleConnected) {
    pTxChar->setValue(msg.c_str());
    pTxChar->notify();
  }
  if (wsClientNum != 255) {
    wsServer.sendTXT(wsClientNum, msg);
  }
  Serial.println("OUT: " + msg);
}

// ─── WebSocket events ─────────────────────────────────────
void onWsEvent(uint8_t num, WStype_t type, uint8_t* payload, size_t length) {
  if (type == WStype_CONNECTED) {
    wsClientNum = num;
    Serial.println("WS client connecté");
  } else if (type == WStype_DISCONNECTED) {
    wsClientNum = 255;
    Serial.println("WS client déconnecté");
  } else if (type == WStype_TEXT) {
    pendingCmd = String((char*)payload);
    cmdPending = true;
  }
}

// ─── BLE Callbacks ────────────────────────────────────────
class ServerCB : public BLEServerCallbacks {
  void onConnect(BLEServer*)    { bleConnected = true; }
  void onDisconnect(BLEServer*) { bleConnected = false; }
};
class RxCB : public BLECharacteristicCallbacks {
  void onWrite(BLECharacteristic* c) {
    String val = c->getValue().c_str();
    val.trim();
    if (val.length() > 0) { pendingCmd = val; cmdPending = true; }
  }
};

// ─── Parser commandes ─────────────────────────────────────
void parseCmd(String cmd) {
  cmd.trim();
  if (cmd.startsWith("MODE:")) {
    currentMode = cmd.substring(5).toInt();
    sendResponse("OK:MODE:" + String(currentMode));
  } else if (cmd.startsWith("BRI:")) {
    int b = cmd.substring(4).toInt();
    if (b >= 80 && b <= 255) { brightness = b; strip.setBrightness(b); }
    sendResponse("OK:BRI:" + String(brightness));
  } else if (cmd.startsWith("COL:")) {
    int p1=cmd.indexOf(':',4), p2=cmd.indexOf(':',p1+1), p3=cmd.indexOf(':',p2+1);
    if (p3 > 0) {
      int idx=cmd.substring(4,p1).toInt();
      if (idx>=0 && idx<=3) {
        pastelR[idx]=constrain(cmd.substring(p1+1,p2).toInt(),0,255);
        pastelG[idx]=constrain(cmd.substring(p2+1,p3).toInt(),0,255);
        pastelB[idx]=constrain(cmd.substring(p3+1).toInt(),0,255);
        sendResponse("OK:COL:"+String(idx));
      }
    }
  } else if (cmd == "OFF")    { currentMode=6; strip.clear(); strip.show(); sendResponse("OK:OFF"); }
  else if (cmd == "RESET")    { currentMode=0; brightness=180; strip.setBrightness(180); sendResponse("OK:RESET"); }
  else if (cmd == "STATUS")   { sendResponse("STATUS:MODE="+String(currentMode)+",BRI="+String(brightness)); }
  else                        { sendResponse("ERR:UNKNOWN:"+cmd); }
}

void setup() {
  Serial.begin(115200);
  strip.begin(); strip.setBrightness(brightness); strip.clear(); strip.show();

  // WiFi
  WiFi.begin(ssid, password);
  Serial.print("WiFi...");
  int t = 0;
  while (WiFi.status() != WL_CONNECTED && t++ < 20) { delay(500); Serial.print("."); }
  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\nIP: " + WiFi.localIP().toString());
    wsServer.begin();
    wsServer.onEvent(onWsEvent);
  }

  // BLE
  BLEDevice::init("LED-Controller");
  pServer = BLEDevice::createServer();
  pServer->setCallbacks(new ServerCB());
  BLEService* svc = pServer->createService(SERVICE_UUID);
  BLECharacteristic* rxC = svc->createCharacteristic(
    CHAR_UUID_RX, BLECharacteristic::PROPERTY_WRITE | BLECharacteristic::PROPERTY_WRITE_NR);
  rxC->setCallbacks(new RxCB());
  pTxChar = svc->createCharacteristic(CHAR_UUID_TX, BLECharacteristic::PROPERTY_NOTIFY);
  pTxChar->addDescriptor(new BLE2902());
  svc->start();
  BLEDevice::getAdvertising()->addServiceUUID(SERVICE_UUID);
  BLEDevice::startAdvertising();

  Serial.println("BLE + WiFi démarrés");
}

void loop() {
  wsServer.loop();

  if (!bleConnected && oldConnected) { delay(300); pServer->startAdvertising(); oldConnected=false; }
  if (bleConnected && !oldConnected)  { oldConnected=true; }

  if (cmdPending) { cmdPending=false; parseCmd(pendingCmd); }

  // Effets LED (identiques à avant)
  if      (currentMode <= 3)  respiration();
  else if (currentMode == 4)  rainbow();
  else if (currentMode == 5)  drapeauFrancais();
  else if (currentMode == 6)  { strip.clear(); strip.show(); }
  else if (currentMode == 7)  vagueFluide();
  else if (currentMode == 10) policeMode();
  else if (currentMode == 11) veilleDiscrete();
  else if (currentMode == 12) fixeStable();
  else if (currentMode == 13) k2000Rouge();
}
