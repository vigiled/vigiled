/*
 * XIAO ESP32C6 — Contrôleur LED NeoPixel
 * BLE (Android) + WiFi WebSocket (iOS)
 * Configuration WiFi via portail captif (WiFiManager)
 *
 * Bibliothèques requises (Gestionnaire de bibliothèques Arduino) :
 *   - Adafruit NeoPixel
 *   - ESP32 Arduino core (Expressif)
 *   - WebSocketsServer  (chercher : "WebSockets" par Links2004)
 *   - WiFiManager       (chercher : "WiFiManager" par tzapu)
 *
 * ─── Première utilisation ────────────────────────────────
 *  1. Flasher ce code
 *  2. L'ESP32 crée un réseau WiFi "LED-Setup"
 *  3. Connectez votre iPhone à "LED-Setup" (Réglages → WiFi)
 *  4. Safari ouvre automatiquement le portail de config
 *  5. Choisissez votre réseau WiFi, entrez le mot de passe
 *  6. L'ESP32 redémarre et se connecte → son IP s'affiche
 *     dans le moniteur série (115200 bauds)
 *  7. Entrez cette IP dans la page web onglet WiFi
 *
 * ─── Réinitialiser le WiFi ───────────────────────────────
 *  Maintenir le bouton MODE (D2) appuyé 5s au démarrage
 *  → efface les credentials et relance le portail
 *
 * Câblage :
 *   NeoPixel      → D1 (GPIO1)
 *   Bouton MODE   → D2 (GPIO2) + GND
 *   Bouton BRIGHT → D3 (GPIO3) + GND
 *   Batterie ADC  → A0 (GPIO0)
 */

#include <Adafruit_NeoPixel.h>
#include <BLEDevice.h>
#include <BLEServer.h>
#include <BLEUtils.h>
#include <BLE2902.h>
#include <WiFi.h>
#include <WiFiManager.h>
#include <WebSocketsServer.h>

// ─── Broches ──────────────────────────────────────────────
#define PIN_NEOPIXEL   1
#define NUM_LEDS       28
#define LEDS_ACTIVE    9
#define BTN_MODE_PIN   2
#define BTN_BRIGHT_PIN 3
#define BAT_PIN        A0

// ─── BLE NUS ──────────────────────────────────────────────
#define SERVICE_UUID   "6E400001-B5BA-F393-E0A9-E50E24DCCA9E"
#define CHAR_UUID_RX   "6E400002-B5BA-F393-E0A9-E50E24DCCA9E"
#define CHAR_UUID_TX   "6E400003-B5BA-F393-E0A9-E50E24DCCA9E"

// ─── Globals ──────────────────────────────────────────────
Adafruit_NeoPixel strip(NUM_LEDS, PIN_NEOPIXEL, NEO_GRB + NEO_KHZ800);
WebSocketsServer  wsServer(81);
uint8_t wsClientNum = 255;

BLEServer*         pServer  = nullptr;
BLECharacteristic* pTxChar  = nullptr;
bool bleConnected = false;
bool oldConnected = false;

uint8_t currentMode = 0;
uint8_t brightness  = 180;
uint8_t pastelR[4] = {  4,  35, 100, 180 };
uint8_t pastelG[4] = { 85,  80,  60,  40 };
uint8_t pastelB[4] = { 42, 160, 120,  20 };

bool   cmdPending = false;
String pendingCmd = "";

// ─── Envoi unifié BLE + WebSocket ─────────────────────────
void sendResponse(String msg) {
  if (pTxChar && bleConnected) {
    pTxChar->setValue(msg.c_str());
    pTxChar->notify();
  }
  if (wsClientNum != 255) wsServer.sendTXT(wsClientNum, msg);
  Serial.println("OUT: " + msg);
}

// ─── WebSocket ────────────────────────────────────────────
void onWsEvent(uint8_t num, WStype_t type, uint8_t* payload, size_t len) {
  if      (type == WStype_CONNECTED)    { wsClientNum = num; Serial.println("WS connecté"); }
  else if (type == WStype_DISCONNECTED) { wsClientNum = 255; Serial.println("WS déconnecté"); }
  else if (type == WStype_TEXT)         { pendingCmd = String((char*)payload); cmdPending = true; }
}

// ─── BLE Callbacks ────────────────────────────────────────
class ServerCB : public BLEServerCallbacks {
  void onConnect(BLEServer*)    { bleConnected = true;  Serial.println("BLE connecté"); }
  void onDisconnect(BLEServer*) { bleConnected = false; Serial.println("BLE déconnecté"); }
};
class RxCB : public BLECharacteristicCallbacks {
  void onWrite(BLECharacteristic* c) {
    String val = c->getValue().c_str(); val.trim();
    if (val.length() > 0) { pendingCmd = val; cmdPending = true; }
  }
};

// ─── Parser commandes ─────────────────────────────────────
void parseCmd(String cmd) {
  cmd.trim();
  if (cmd.startsWith("MODE:")) {
    currentMode = cmd.substring(5).toInt();
    sendResponse("OK:MODE:" + String(currentMode));
  } else if (cmd.startsWith("BRI:")) {
    int b = cmd.substring(4).toInt();
    if (b >= 80 && b <= 255) { brightness = b; strip.setBrightness(b); }
    sendResponse("OK:BRI:" + String(brightness));
  } else if (cmd.startsWith("COL:")) {
    int p1=cmd.indexOf(':',4), p2=cmd.indexOf(':',p1+1), p3=cmd.indexOf(':',p2+1);
    if (p3 > 0) {
      int idx = cmd.substring(4, p1).toInt();
      if (idx >= 0 && idx <= 3) {
        pastelR[idx] = constrain(cmd.substring(p1+1, p2).toInt(), 0, 255);
        pastelG[idx] = constrain(cmd.substring(p2+1, p3).toInt(), 0, 255);
        pastelB[idx] = constrain(cmd.substring(p3+1).toInt(),     0, 255);
        sendResponse("OK:COL:" + String(idx));
      }
    }
  } else if (cmd == "OFF")  { currentMode=6; strip.clear(); strip.show(); sendResponse("OK:OFF"); }
  else if (cmd == "RESET")  { currentMode=0; brightness=180; strip.setBrightness(180); sendResponse("OK:RESET"); }
  else if (cmd == "STATUS") { sendResponse("STATUS:MODE="+String(currentMode)+",BRI="+String(brightness)); }
  else                      { sendResponse("ERR:UNKNOWN:"+cmd); }
}

// ─── LED pendant config WiFi ──────────────────────────────
void ledConfigPortal() {
  // Pulsation orange = en attente de configuration
  static unsigned long t = 0;
  if (millis() - t > 20) {
    t = millis();
    float bri = (sinf(millis() / 600.0f) + 1.0f) / 2.0f;
    uint8_t v = (uint8_t)(bri * 120);
    strip.fill(strip.Color(v, v/3, 0), 0, LEDS_ACTIVE);
    strip.show();
  }
}

// ─── SETUP ────────────────────────────────────────────────
void setup() {
  Serial.begin(115200);
  pinMode(BTN_MODE_PIN,   INPUT_PULLUP);
  pinMode(BTN_BRIGHT_PIN, INPUT_PULLUP);

  strip.begin();
  strip.setBrightness(brightness);
  strip.clear();
  strip.show();

  // ── Réinitialisation WiFi si bouton MODE tenu au boot ──
  if (digitalRead(BTN_MODE_PIN) == LOW) {
    Serial.println("Reset WiFi demandé...");
    strip.fill(strip.Color(100, 0, 0), 0, LEDS_ACTIVE); strip.show();
    delay(3000);
    if (digitalRead(BTN_MODE_PIN) == LOW) {
      WiFiManager wm;
      wm.resetSettings();
      Serial.println("Credentials WiFi effacés !");
      strip.fill(strip.Color(0, 100, 0), 0, LEDS_ACTIVE); strip.show();
      delay(1000);
    }
  }

  // ── WiFiManager : portail auto si pas de réseau connu ──
  WiFiManager wm;
  wm.setAPName("LED-Setup");           // Nom du réseau de config
  wm.setAPStaticIPConfig(
    IPAddress(192,168,4,1),            // IP du portail
    IPAddress(192,168,4,1),
    IPAddress(255,255,255,0)
  );

  // Callback pendant que le portail est actif
  wm.setAPCallback([](WiFiManager* wm) {
    Serial.println("Portail de config actif — connectez-vous à LED-Setup");
    Serial.println("Puis ouvrez http://192.168.4.1");
  });

  // Tentative de connexion (timeout 3min puis portail)
  wm.setConfigPortalTimeout(180);
  bool connected = wm.autoConnect("LED-Setup", "");  // sans mdp sur le réseau de config

  if (!connected) {
    Serial.println("Timeout config — redémarrage");
    ESP.restart();
  }

  // ── WiFi connecté ─────────────────────────────────────
  Serial.println("\n✓ WiFi connecté !");
  Serial.println("IP: " + WiFi.localIP().toString());

  // Clignotement vert = WiFi OK
  for (int i = 0; i < 3; i++) {
    strip.fill(strip.Color(0, 120, 40), 0, LEDS_ACTIVE); strip.show(); delay(200);
    strip.clear(); strip.show(); delay(150);
  }

  // Démarrer WebSocket
  wsServer.begin();
  wsServer.onEvent(onWsEvent);
  Serial.println("WebSocket démarré sur port 81");

  // ── BLE ───────────────────────────────────────────────
  BLEDevice::init("LED-Controller");
  pServer = BLEDevice::createServer();
  pServer->setCallbacks(new ServerCB());
  BLEService* svc = pServer->createService(SERVICE_UUID);
  BLECharacteristic* rxC = svc->createCharacteristic(
    CHAR_UUID_RX,
    BLECharacteristic::PROPERTY_WRITE | BLECharacteristic::PROPERTY_WRITE_NR
  );
  rxC->setCallbacks(new RxCB());
  pTxChar = svc->createCharacteristic(CHAR_UUID_TX, BLECharacteristic::PROPERTY_NOTIFY);
  pTxChar->addDescriptor(new BLE2902());
  svc->start();
  BLEDevice::getAdvertising()->addServiceUUID(SERVICE_UUID);
  BLEDevice::startAdvertising();

  Serial.println("BLE démarré : LED-Controller");
  Serial.println("─── Tout est prêt ! ───");
}

// ─── LOOP ────────────────────────────────────────────────
void loop() {
  wsServer.loop();

  // Reconnexion BLE auto
  if (!bleConnected && oldConnected) {
    delay(300);
    pServer->startAdvertising();
    oldConnected = false;
  }
  if (bleConnected && !oldConnected) oldConnected = true;

  // Traiter commande entrante
  if (cmdPending) { cmdPending = false; parseCmd(pendingCmd); }

  // Effets LED
  if      (currentMode <= 3)  respiration();
  else if (currentMode == 4)  rainbow();
  else if (currentMode == 5)  drapeauFrancais();
  else if (currentMode == 6)  { strip.clear(); strip.show(); delay(50); }
  else if (currentMode == 7)  vagueFluide();
  else if (currentMode == 10) policeMode();
  else if (currentMode == 11) veilleDiscrete();
  else if (currentMode == 12) fixeStable();
  else if (currentMode == 13) k2000Rouge();
}
